```js
// src/js/auth.js
import { supabase } from './supabase.js'

/**
 * Redirects to admin login if no active session.
 * Call at the top of every admin page script.
 */
export async function requireAuth() {
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) {
    window.location.href = '/admin/index.html'
    return null
  }
  return session
}

/**
 * Returns current session or null (no redirect).
 */
export async function getSession() {
  const { data: { session } } = await supabase.auth.getSession()
  return session
}

export async function signOut() {
  await supabase.auth.signOut()
  window.location.href = '/admin/index.html'
}
```

```js
// src/js/formspree.js
import { supabase } from './supabase.js'

const FORMSPREE_ID = import.meta.env.VITE_FORMSPREE_ID

export function initContactForm() {
  const form = document.querySelector('.contato-form')
  if (!form) return

  form.addEventListener('submit', async (e) => {
    e.preventDefault()

    const btn = form.querySelector('.form-submit')
    const originalText = btn.textContent
    btn.textContent = 'Enviando...'
    btn.disabled = true

    const data = {
      nome: form.querySelector('[name="nome"]').value.trim(),
      telefone: form.querySelector('[name="telefone"]').value.trim(),
      email: form.querySelector('[name="email"]').value.trim(),
      perfil: form.querySelector('[name="perfil"]').value,
      mensagem: form.querySelector('[name="mensagem"]').value.trim(),
    }

    // Validate required fields
    if (!data.nome || !data.email || !data.mensagem) {
      showFormFeedback(form, 'Por favor, preencha nome, email e mensagem.', 'error')
      btn.textContent = originalText
      btn.disabled = false
      return
    }

    try {
      // Save to Supabase + send email via Formspree in parallel
      const [supaResult, formspreeResult] = await Promise.allSettled([
        supabase.from('contato_mensagem').insert([data]),
        fetch(`https://formspree.io/f/${FORMSPREE_ID}`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(data),
        }),
      ])

      // At least one should succeed
      const anySuccess = supaResult.status === 'fulfilled' && !supaResult.value.error
        || formspreeResult.status === 'fulfilled' && formspreeResult.value.ok

      if (anySuccess) {
        showFormFeedback(form, 'Mensagem enviada! Retornaremos em breve.', 'success')
        form.reset()
      } else {
        throw new Error('Ambos os envios falharam')
      }
    } catch (err) {
      console.error('Contact form error:', err)
      showFormFeedback(form, 'Erro ao enviar. Tente novamente ou entre em contato por WhatsApp.', 'error')
    } finally {
      btn.textContent = originalText
      btn.disabled = false
    }
  })
}

function showFormFeedback(form, message, type) {
  let el = form.querySelector('.form-feedback')
  if (!el) {
    el = document.createElement('div')
    el.className = 'form-feedback'
    form.appendChild(el)
  }
  el.textContent = message
  el.className = `form-feedback form-feedback--${type}`
  setTimeout(() => el.remove(), 6000)
}
```

```js
  // First load the project to get its ID
  const { data: p, error } = await supabase.from('projeto').select('*').eq('slug', slug).single()

  if (error || !p) {
    content.innerHTML = '<div class="projeto-loading">Projeto não encontrado.</div>'
    return
  }

  // Then load related data in parallel with the real projeto_id
  const [imgs, units, updates] = await Promise.all([
    supabase.from('projeto_imagem').select('*').eq('projeto_id', p.id).order('sort_order'),
    supabase.from('projeto_unidade').select('*').eq('projeto_id', p.id).order('identificador'),
    supabase.from('projeto_update').select('*').eq('projeto_id', p.id).order('data', { ascending: false }),
  ])

  // Update page title and OG tags
  document.title = `${p.nome} — Pilar360`
  document.getElementById('og-title')?.setAttribute('content', p.nome)
  document.getElementById('og-desc')?.setAttribute('content', p.descricao?.substring(0, 160) || '')
  document.getElementById('og-image')?.setAttribute('content', p.imagem_capa_url || '')

  content.innerHTML = `
    ${renderHero(p)}
    ${renderSpecs(p)}
    ${renderDescricao(p, imgs.data || [])}
    ${p.planta_url ? renderPlanta(p) : ''}
    ${renderUnidades(units.data || [])}
    ${renderSimulador(p)}
    ${renderTimeline(updates.data || [])}
    ${renderCTA(p)}
  `

  // Init lightbox, simulador, fade-in after render
  initLightbox()
  initSimulador(p)
  initFadeIn()
}

function initSimulador(projeto) {
  const btn = document.getElementById('sim-calcular')
  if (!btn) return

  btn.addEventListener('click', () => {
    const rendaStr = document.getElementById('sim-renda').value.replace(/\D/g, '')
    const valorStr = document.getElementById('sim-valor').value.replace(/\D/g, '')
    const renda = parseInt(rendaStr) / 100 || 0
    const valor = parseInt(valorStr) / 100 || 0

    if (renda <= 0 || valor <= 0) {
      document.getElementById('sim-resultado').style.display = 'block'
      document.getElementById('sim-resultado').innerHTML = '<p class="sim-error">Preencha renda e valor do imóvel.</p>'
      return
    }

    // MCMV 2024/2025 simplified brackets
    let faixa, subsidio, taxaJuros
    if (renda <= 2640) {
      faixa = 'Faixa 1'
      subsidio = Math.min(valor * 0.50, 55000)
      taxaJuros = 0.04
    } else if (renda <= 4400) {
      faixa = 'Faixa 2'
      subsidio = Math.min(valor * 0.25, 35000)
      taxaJuros = 0.055
    } else if (renda <= 8000) {
      faixa = 'Faixa 3'
      subsidio = 0
      taxaJuros = 0.075
    } else {
      faixa = 'Acima do MCMV'
      subsidio = 0
      taxaJuros = 0.09
    }

    const financiado = valor - subsidio
    const entrada = financiado * 0.20
    const principal = financiado - entrada
    const meses = 420 // 35 years
    const taxaMensal = taxaJuros / 12
    const parcela = principal * (taxaMensal * Math.pow(1 + taxaMensal, meses)) / (Math.pow(1 + taxaMensal, meses) - 1)

    const fmt = (v) => v.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' })

    const whatsMsg = encodeURIComponent(
      `Olá! Simulei o financiamento do ${projeto.nome} e gostaria de mais informações.\n` +
      `Renda: ${fmt(renda)}\nFaixa: ${faixa}\nValor: ${fmt(valor)}\nParcela estimada: ${fmt(parcela)}`
    )

    document.getElementById('sim-resultado').style.display = 'block'
    document.getElementById('sim-resultado').innerHTML = `
      <div class="sim-grid">
        <div class="sim-item"><span class="sim-label">Faixa</span><span class="sim-val">${faixa}</span></div>
        <div class="sim-item"><span class="sim-label">Subsídio estimado</span><span class="sim-val">${fmt(subsidio)}</span></div>
        <div class="sim-item"><span class="sim-label">Entrada (~20%)</span><span class="sim-val">${fmt(entrada)}</span></div>
        <div class="sim-item"><span class="sim-label">Parcela estimada</span><span class="sim-val sim-val--destaque">${fmt(parcela)}</span></div>
      </div>
      <p class="sim-disclaimer">* Simulação aproximada. Valores sujeitos a análise de crédito.</p>
      <a href="https://wa.me/5551992143280?text=${whatsMsg}" target="_blank" class="btn-primary" style="margin-top:1.5rem;">Falar com um consultor</a>
    `
  })
}

function initFadeIn() {
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) entry.target.classList.add('visible')
    })
  }, { threshold: 0.1 })
  document.querySelectorAll('.fade-in').forEach(el => observer.observe(el))
}
```

```js
// src/js/blog.js
import { renderHeader } from './header.js'
import { renderFooter } from './footer.js'
import { supabase } from './supabase.js'

renderHeader()
renderFooter()

const PAGE_SIZE = 20
let currentCat = 'todos'
let offset = 0
let allLoaded = false

async function loadPosts(reset = false) {
  if (reset) {
    offset = 0
    allLoaded = false
  }

  const grid = document.getElementById('blog-grid')
  if (reset) grid.innerHTML = ''

  let query = supabase
    .from('blog_post')
    .select('id, titulo, slug, resumo, categoria, imagem_capa_url, publicado_em')
    .eq('publicado', true)
    .order('publicado_em', { ascending: false })
    .range(offset, offset + PAGE_SIZE - 1)

  if (currentCat !== 'todos') {
    query = query.eq('categoria', currentCat)
  }

  const { data, error } = await query

  if (error || !data) {
    grid.innerHTML = '<p class="section-text">Erro ao carregar posts.</p>'
    return
  }
```

```js
async function loadPost() {
  const content = document.getElementById('post-content')

  const { data: post, error } = await supabase
    .from('blog_post')
    .select('*')
    .eq('slug', slug)
    .eq('publicado', true)
```

```js
function toSlug(text) {
  return text.toLowerCase()
    .normalize('NFD').replace(/[\u0300-\u036f]/g, '')
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '')
}
```
